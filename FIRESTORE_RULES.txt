// Firestore Security Rules for Hornet Mapper (dev-friendly)
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow read/write in the 4 subcollections under the 'data' doc
    match /maps/{year}/{group}/data/markers/{docId} { allow read, write: if true; }
    match /maps/{year}/{group}/data/lines/{docId}   { allow read, write: if true; }
    match /maps/{year}/{group}/data/sectors/{docId} { allow read, write: if true; }
    match /maps/{year}/{group}/data/polygons/{docId}{ allow read, write: if true; }
  }
}
